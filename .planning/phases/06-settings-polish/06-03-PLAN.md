Korrekte Implementierung mit @State (da supabase.auth.session async ist):
```swift
// In RootView:
@EnvironmentObject private var authService: AuthService
@EnvironmentObject private var familyService: FamilyService
@State private var currentMemberRole: String? = nil

// Im body, case .authenticated(hasFamily: true):
case .authenticated(hasFamily: true):
    Group {
        if currentMemberRole == MemberRole.child.rawValue {
            KindDashboardView()
        } else {
            // MainTabView (Phase 4) oder AuthenticatedPlaceholderView (Uebergangsloesung)
            AuthenticatedPlaceholderView()
                .environmentObject(familyService)
        }
    }
    .task {
        // Eingeloggte User-ID aus Supabase-Session lesen
        if let userId = try? await supabase.auth.session.user.id {
            // Member mit dieser ID in familyService.members finden
            if let member = familyService.members.first(where: { $0.id == userId }) {
                currentMemberRole = member.role
            } else {
                // Members noch nicht geladen — lade sie
                if let familyId = familyService.currentFamily?.id {
                    await familyService.fetchMembers(familyId: familyId)
                    if let member = familyService.members.first(where: { $0.id == userId }) {
                        currentMemberRole = member.role
                    }
                }
            }
        }
    }
```
