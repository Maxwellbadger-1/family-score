**fastlane/Fastfile:**
```ruby
# fastlane/Fastfile
# Verwendet: App Store Connect API Key (kein Apple-ID/2FA in CI)
# Pitfall 5 mitigiert: beide Targets in export_options.provisioningProfiles

default_platform(:ios)

platform :ios do
  lane :release_appstore do
    # 1. App Store Connect API Key (kein 2FA in CI)
    app_store_connect_api_key(
      key_id: ENV["ASC_KEY_ID"],
      issuer_id: ENV["ASC_ISSUER_ID"],
      key_content: ENV["ASC_KEY"],
      in_house: false
    )

    # 2. Temporaere Keychain fuer CI (Pflicht in GitHub Actions)
    setup_ci

    # 3. Verteilungs-Zertifikate + Profile aus Zertifikats-Repo
    match(
      type: "appstore",
      readonly: is_ci,
      git_url: ENV["MATCH_GIT_URL"],
      app_identifier: ["com.familyscore", "com.familyscore.widget"]
    )

    # 4. Release-Build mit korrektem Export (Pitfall 5: beide Targets)
    build_app(
      project: "FamilyScore/FamilyScore.xcodeproj",
      scheme: "FamilyScore",
      configuration: "Release",
      export_method: "app-store",
      cloned_source_packages_path: "~/spm-packages",
      export_options: {
        provisioningProfiles: {
          "com.familyscore" => "match AppStore com.familyscore",
          "com.familyscore.widget" => "match AppStore com.familyscore.widget"
        }
      }
    )

    # 5. Upload zu TestFlight
    upload_to_testflight(skip_waiting_for_build_processing: true)
  end
end
```

```yaml
# .github/workflows/release.yml
# Manueller App-Store-Release-Workflow — wird NICHT bei Push ausgefuehrt.
# Trigger: GitHub Actions → Workflows → "App Store Release" → "Run workflow"
# Benoetigt GitHub Actions Secrets:
#   - SUPABASE_URL_SECRET, SUPABASE_KEY_SECRET (wie in build.yml)
#   - ASC_KEY_ID, ASC_ISSUER_ID, ASC_KEY (.p8-Inhalt als String)
#   - MATCH_GIT_URL (HTTPS-URL zum privaten Zertifikats-Repo)
#   - MATCH_PASSWORD (Verschluesselungspasswort fuer match)
#   - MATCH_GIT_BASIC_AUTHORIZATION (base64("username:PAT"))
#   - APPLE_ID (Apple Developer Account E-Mail)
#   - APPLE_TEAM_ID (10-stelliger Team-Identifier)
#   - ITC_TEAM_ID (App Store Connect Team-Nummer)

name: App Store Release
on:
  workflow_dispatch:    # Nur manueller Trigger

env:
  FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: true

jobs:
  release:
    runs-on: macos-15
    steps:
      - uses: actions/checkout@v4

      - name: Select Xcode 16
        run: |
          XCODE=$(ls /Applications/ | grep "^Xcode_16" | sort -V | tail -1)
          sudo xcode-select -s "/Applications/$XCODE/Contents/Developer"
          xcodebuild -version

      - name: Setup Ruby
        uses: ruby/setup-ruby@v1
        with:
          ruby-version: '3.4'
          bundler-cache: true

      - name: Install XcodeGen + xcpretty
        run: brew install xcodegen && gem install xcpretty --no-document

      - name: Create Secrets.xcconfig
        run: |
          SUPABASE_HOST=$(echo "${{ secrets.SUPABASE_URL_SECRET }}" | sed 's|https://||')
          cat > FamilyScore/Config/Secrets.xcconfig << EOF
          SUPABASE_URL_SECRET = $SUPABASE_HOST
          SUPABASE_KEY_SECRET = ${{ secrets.SUPABASE_KEY_SECRET }}
          EOF

      - name: Generate Xcode Project
        working-directory: FamilyScore
        run: xcodegen generate --spec project.yml

      - name: Cache SPM packages
        uses: actions/cache@v4
        with:
          path: ~/spm-packages
          key: ${{ runner.os }}-spm-${{ hashFiles('FamilyScore/project.yml') }}
          restore-keys: |
            ${{ runner.os }}-spm-

      - name: Resolve SPM dependencies
        working-directory: FamilyScore
        run: |
          xcodebuild -resolvePackageDependencies \
            -scheme FamilyScore \
            -clonedSourcePackagesDirPath ~/spm-packages

      - name: Run fastlane release_appstore
        run: bundle exec fastlane release_appstore
        env:
          ASC_KEY_ID: ${{ secrets.ASC_KEY_ID }}
          ASC_ISSUER_ID: ${{ secrets.ASC_ISSUER_ID }}
          ASC_KEY: ${{ secrets.ASC_KEY }}
          MATCH_PASSWORD: ${{ secrets.MATCH_PASSWORD }}
          MATCH_GIT_URL: ${{ secrets.MATCH_GIT_URL }}
          MATCH_GIT_BASIC_AUTHORIZATION: ${{ secrets.MATCH_GIT_BASIC_AUTHORIZATION }}
          APPLE_ID: ${{ secrets.APPLE_ID }}
          APPLE_TEAM_ID: ${{ secrets.APPLE_TEAM_ID }}
          ITC_TEAM_ID: ${{ secrets.ITC_TEAM_ID }}
```
