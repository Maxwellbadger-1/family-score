```swift
// FamilyScoreKit/Sources/FamilyScoreKit/Intents/LogActivityIntent.swift
// Target Membership: FamilyScoreKit (shared package)
// KRITISCH: Sowohl App-Target als auch Widget Extension brauchen diesen Intent
//           Loesung: FamilyScoreKit = einzige Quelle (Pitfall 8 aus 05-RESEARCH.md)
// KEIN Supabase-Import — perform() schreibt nur in App Group (30MB Widget-Limit)

import AppIntents
import Foundation

@available(iOS 16.0, *)
public struct LogActivityIntent: AppIntent {
    public static let title: LocalizedStringResource = "Aktivitaet eintragen"

    @Parameter(title: "Kategorie")
    public var categorySlug: String

    public init() {
        self.categorySlug = "haushalt"  // Default-Wert (required fuer AppIntent)
    }

    public init(categorySlug: String) {
        self.categorySlug = categorySlug
    }

    // perform() wird im Widget-Prozess ausgefuehrt
    // NIEMALS Supabase oder Netzwerkcall hier — 30MB Widget-Prozess-Limit
    // Strategie: pending_log in App Group; App drainiert beim naechsten Foreground
    public func perform() async throws -> some IntentResult {
        guard let defaults = UserDefaults(suiteName: appGroupIdentifier) else {
            return .result()
        }
        var pending = (defaults.array(forKey: "pendingLogs") as? [[String: String]]) ?? []
        pending.append([
            "category": categorySlug,
            "timestamp": ISO8601DateFormatter().string(from: .now)
        ])
        defaults.set(pending, forKey: "pendingLogs")
        return .result()
    }
}
```

```swift
// Drainiert pending_logs die LogActivityIntent.perform() in die App Group geschrieben hat
// Wird bei jedem scenePhase == .active aufgerufen (WIDGET-04 Integration)
private func drainPendingWidgetLogs() async {
    guard let defaults = UserDefaults(suiteName: appGroupIdentifier),
          let pending = defaults.array(forKey: "pendingLogs") as? [[String: String]],
          !pending.isEmpty
    else { return }

    // Atomar: zuerst loeschen, dann verarbeiten (verhindert Doppel-Verarbeitung)
    defaults.removeObject(forKey: "pendingLogs")

    for log in pending {
        guard let slug = log["category"] else { continue }
        // Timestamp aus pending_log (ISO8601) fuer retroaktiven Eintrag
        let timestamp = log["timestamp"].flatMap { ISO8601DateFormatter().date(from: $0) } ?? .now
        let minutesSince = max(1, Int(Date.now.timeIntervalSince(timestamp) / 60))
        // Navigation: NotificationCenter informiert die Activity-View ueber Widget-Log
        NotificationCenter.default.post(
            name: .openQuickLog,
            object: nil,
            userInfo: [
                "category": slug,
                "duration": minutesSince,
                "fromWidget": true
            ]
        )
    }
}
```
