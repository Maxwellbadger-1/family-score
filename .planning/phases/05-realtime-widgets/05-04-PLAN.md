```sql
-- =============================================================================
-- Family Score: Phase 5 Push Tokens
-- device_push_tokens Tabelle fuer APNs opt-in Push-Benachrichtigungen (SYNC-03)
-- Migration: 20260517_phase5_push_tokens.sql
-- Sicherheit: RLS — User verwaltet nur eigene Token
--             Edge Function liest via Service-Role (umgeht RLS) — keine zusaetzliche Policy
-- =============================================================================

CREATE TABLE public.device_push_tokens (
  id          uuid primary key default gen_random_uuid(),
  user_id     uuid not null references auth.users(id) on delete cascade,
  family_id   uuid not null references public.families(id) on delete cascade,
  apns_token  text not null,
  device_name text,
  created_at  timestamptz not null default now(),
  updated_at  timestamptz not null default now(),
  UNIQUE (user_id, apns_token)
);

-- Index fuer schnelle Abfrage aller Tokens einer Familie (Edge Function)
CREATE INDEX device_push_tokens_family_id_idx ON public.device_push_tokens(family_id);
CREATE INDEX device_push_tokens_user_id_idx ON public.device_push_tokens(user_id);

ALTER TABLE public.device_push_tokens ENABLE ROW LEVEL SECURITY;

-- User verwaltet eigene Token (INSERT + SELECT + UPDATE + DELETE)
-- (select auth.uid()) statt auth.uid() — Performance-Best-Practice aus RESEARCH.md
CREATE POLICY "User verwaltet eigene Push-Tokens"
  ON public.device_push_tokens FOR ALL TO authenticated
  USING ((SELECT auth.uid()) = user_id)
  WITH CHECK ((SELECT auth.uid()) = user_id);

-- Edge Function liest alle Familien-Token via Service-Role-Client (umgeht RLS)
-- Keine zusaetzliche SELECT-Policy fuer andere Users noetig — Service-Role ist ausreichend
-- KEIN client-seitiger Zugriff auf fremde Tokens moeglich
```

```swift
// FamilyScore/Services/NotificationService.swift
// Target Membership: FamilyScore (App) ONLY
// iOS 16.0 Minimum: ObservableObject + @Published (NICHT @Observable)
// Opt-in Push: User muss aktiv zustimmen — KEIN automatischer Request beim App-Start

import Foundation
import UserNotifications

@MainActor
final class NotificationService: ObservableObject {

    @Published private(set) var isAuthorized: Bool = false
    @Published private(set) var authorizationStatus: UNAuthorizationStatus = .notDetermined

    // MARK: - Permission Request (opt-in)

    /// Zeigt System-Prompt fuer Push-Benachrichtigungen.
    /// Wird NUR aufgerufen wenn User aktiv zustimmt (NotificationPermissionView).
    func requestPermission() async {
        let center = UNUserNotificationCenter.current()
        do {
            let granted = try await center.requestAuthorization(options: [.alert, .sound, .badge])
            isAuthorized = granted
            if granted {
                // registerForRemoteNotifications muss auf Main Thread aufgerufen werden
                await MainActor.run {
                    UIApplication.shared.registerForRemoteNotifications()
                }
            }
        } catch {
            isAuthorized = false
        }
        // Status nach Entscheidung aktualisieren
        await refreshAuthorizationStatus()
    }

    func refreshAuthorizationStatus() async {
        let center = UNUserNotificationCenter.current()
        let settings = await center.notificationSettings()
        authorizationStatus = settings.authorizationStatus
        isAuthorized = settings.authorizationStatus == .authorized
    }

    // MARK: - APNs Token speichern

    /// Wird von AppDelegate/FamilyScoreApp nach didRegisterForRemoteNotificationsWithDeviceToken aufgerufen.
    /// Speichert APNs-Token als Hex-String in device_push_tokens via upsert.
    /// NIEMALS APNs-Token in App Group speichern — nur in Supabase DB (Security: 05-RESEARCH.md)
    func saveToken(_ deviceToken: Data, familyId: UUID, userId: UUID) async {
        // Data → Hex-String (Standard iOS APNs Token Format)
        let tokenString = deviceToken.map { String(format: "%02.2hhx", $0) }.joined()

        struct PushToken: Encodable {
            let user_id: String
            let family_id: String
            let apns_token: String
        }

        try? await supabase
            .from("device_push_tokens")
            .upsert(
                PushToken(
                    user_id: userId.uuidString,
                    family_id: familyId.uuidString,
                    apns_token: tokenString
                ),
                onConflict: "user_id,apns_token"
            )
            .execute()
    }
}
```

```typescript
// supabase/functions/push-notification/index.ts
// Supabase Edge Function — Deno Runtime
// Trigger: Database Webhook auf activity_entries INSERT
// Zweck: APNs Push-Benachrichtigung an alle anderen Familienmitglieder senden (SYNC-03)
//
// SPIKE-STATUS (Assumption A1 aus 05-RESEARCH.md):
// APNs JWT-Signierung via jose in Deno — Community-Reports gemischt.
// Diese Implementierung folgt dem dokumentierten Standard-Ansatz.
// Falls jose-Import fehlschlaegt: Fallback-Evaluation (Expo Push Service) dokumentieren.

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'
import { SignJWT, importPKCS8 } from 'https://esm.sh/jose@5'

// Typen fuer den Supabase Database Webhook Payload
interface WebhookPayload {
  type: 'INSERT';
  table: 'activity_entries';
  record: {
    id: string;
    family_id: string;
    user_id: string;
    duration_minutes: number;
    points: number;
  };
}

interface PushToken {
  apns_token: string;
  user_id: string;
}

// Service-Role-Client: umgeht RLS, um alle Familien-Tokens zu lesen
// Service-Role-Key darf NIEMALS im iOS-Bundle sein — nur in Edge Function Secrets
const supabaseAdmin = createClient(
  Deno.env.get('SUPABASE_URL') ?? '',
  Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? '',
  { auth: { persistSession: false } }
)

Deno.serve(async (req: Request) => {
  try {
    const payload: WebhookPayload = await req.json()

    if (payload.type !== 'INSERT' || payload.table !== 'activity_entries') {
      return new Response('Ignored', { status: 200 })
    }

    const { family_id, user_id } = payload.record

    // Alle anderen Familienmitglieder-Tokens laden (eigenes Geraet ausschliessen)
    const { data: tokens, error } = await supabaseAdmin
      .from('device_push_tokens')
      .select('apns_token, user_id')
      .eq('family_id', family_id)
      .neq('user_id', user_id)  // Eigenes Geraet nicht benachrichtigen

    if (error) {
      console.error('Error loading tokens:', error.message)
      return new Response('DB Error', { status: 500 })
    }

    if (!tokens || tokens.length === 0) {
      return new Response('No tokens', { status: 200 })
    }

    // APNs JWT erzeugen (jose + p8 Key aus Supabase Secrets)
    // SPIKE: Falls jose/importPKCS8 in Deno fehlschlaegt → Assumption A1 bestaetigt
    const apnsToken = await generateApnsJwt()

    // Push an alle anderen Familienmitglieder senden
    const results = await Promise.allSettled(
      (tokens as PushToken[]).map(({ apns_token }) =>
        sendApnsNotification(apns_token, apnsToken, {
          aps: {
            alert: {
              title: 'Neue Aktivitaet',
              body: 'Ein Familienmitglied hat etwas eingetragen!'
            },
            'content-available': 1
          }
        })
      )
    )

    const failed = results.filter(r => r.status === 'rejected').length
    if (failed > 0) {
      console.warn(`${failed}/${tokens.length} push notifications failed`)
    }

    return new Response(`Sent to ${tokens.length - failed}/${tokens.length} devices`, { status: 200 })

  } catch (error) {
    console.error('Edge function error:', error)
    return new Response('Internal Error', { status: 500 })
  }
})

// APNs JWT erzeugen via jose (p8 Key)
// Quellen: APNs HTTP/2 Auth mit JWT (Apple Docs), jose ESM (esm.sh)
async function generateApnsJwt(): Promise<string> {
  const p8Key = Deno.env.get('APNS_P8_KEY') ?? ''
  const keyId = Deno.env.get('APNS_KEY_ID') ?? ''
  const teamId = Deno.env.get('APNS_TEAM_ID') ?? ''

  // p8 Key als PKCS8 importieren
  const privateKey = await importPKCS8(p8Key, 'ES256')

  return new SignJWT({})
    .setProtectedHeader({ alg: 'ES256', kid: keyId })
    .setIssuedAt()
    .setIssuer(teamId)
    .sign(privateKey)
}

// APNs HTTP/2 Push senden
async function sendApnsNotification(
  deviceToken: string,
  jwtToken: string,
  payload: object
): Promise<void> {
  const bundleId = Deno.env.get('APNS_BUNDLE_ID') ?? 'com.familyscore'
  // Produktion: api.push.apple.com; Entwicklung: api.sandbox.push.apple.com
  const apnsUrl = `https://api.sandbox.push.apple.com/3/device/${deviceToken}`

  const response = await fetch(apnsUrl, {
    method: 'POST',
    headers: {
      'authorization': `bearer ${jwtToken}`,
      'apns-topic': bundleId,
      'apns-push-type': 'alert',
      'content-type': 'application/json'
    },
    body: JSON.stringify(payload)
  })

  if (!response.ok) {
    const body = await response.text()
    throw new Error(`APNs error ${response.status}: ${body}`)
  }
}
```
