// FamilyScore/Services/RealtimeService.swift
// Target Membership: FamilyScore (App) ONLY — NIEMALS Widget Extension
// iOS 16.0 Minimum: ObservableObject + @Published (NICHT @Observable)
// Supabase SDK NUR im Hauptapp-Target (CLAUDE.md Architektur-Regel)

import Foundation
@preconcurrency import Supabase

@MainActor
final class RealtimeService: ObservableObject {

    @Published private(set) var isListening: Bool = false

    private var channel: RealtimeChannelV2?
    private var listeningTask: Task<Void, Never>?

    // Schwache Referenzen auf ActivityService werden bei Wave 2 App-Verdrahtung injiziert
    // Phase 5 Wave 1: Callback-basiert via NotificationCenter
    var onActivityInserted: ((ActivityEntry) -> Void)?
    var onActivityDeleted: ((UUID) -> Void)?

    // MARK: - Starten (scenePhase == .active)

    func startListening(familyId: UUID) async {
        // KRITISCH: Dieser Guard MUSS die ERSTE Zeile sein.
        // Ohne Guard: Supabase Realtime haengt 227s in CI und crasht mit signal trap
        // Quelle: CLAUDE.md "XCTest / Test-Architektur (Regeln aus wiederholten Fehlern)"
        guard ProcessInfo.processInfo.environment["XCTestConfigurationFilePath"] == nil else {
            print("[Realtime] XCTest-Umgebung erkannt — startListening() uebersprungen")
            return
        }

        // Idempotenz: wenn bereits verbunden, nichts tun
        if channel?.status == .subscribed { return }

        // Cleanup alten Channel (mit 100ms Pause um SUBSCRIBED/CLOSED-Loop zu vermeiden)
        // Bekannter Supabase-Swift-Bug: GitHub Discussion #27513
        await stopListening()
        try? await Task.sleep(nanoseconds: 100_000_000)  // 100ms

        let ch = await supabase.channel("family-\(familyId.uuidString)")

        // ACHTUNG: postgresChange() MUSS vor subscribe() aufgerufen werden
        let entryChanges = await ch.postgresChange(
            AnyAction.self,
            schema: "public",
            table: "activity_entries",
            filter: .eq("family_id", value: familyId.uuidString)
        )

        let summaryChanges = await ch.postgresChange(
            AnyAction.self,
            schema: "public",
            table: "weekly_summaries",
            filter: .eq("family_id", value: familyId.uuidString)
        )

        await ch.subscribe()
        self.channel = ch
        isListening = true

        listeningTask = Task {
            await withTaskGroup(of: Void.self) { group in
                group.addTask {
                    for await change in entryChanges {
                        await self.handleActivityChange(change)
                    }
                }
                group.addTask {
                    for await change in summaryChanges {
                        await self.handleSummaryChange(change)
                    }
                }
            }
        }
    }

    // MARK: - Stoppen (scenePhase == .background)

    func stopListening() async {
        listeningTask?.cancel()
        listeningTask = nil
        if let ch = channel {
            await supabase.removeChannel(ch)
            channel = nil
        }
        isListening = false
    }

    // MARK: - Reconnect nach Foreground (REST-Fetch zuerst, dann re-subscribe)

    func refetchAndResubscribe(familyId: UUID) async {
        // 100ms Pause nach removeChannel (Pitfall 2: SUBSCRIBED/CLOSED-Loop)
        await stopListening()
        try? await Task.sleep(nanoseconds: 100_000_000)
        await startListening(familyId: familyId)
    }

    // MARK: - Event-Handler

    private func handleActivityChange(_ change: AnyAction) async {
        switch change {
        case .insert(let action):
            if let entry = try? action.decodeRecord(as: ActivityEntry.self) {
                onActivityInserted?(entry)
            }
        case .delete(let action):
            if let idString = action.oldRecord["id"]?.stringValue,
               let id = UUID(uuidString: idString) {
                onActivityDeleted?(id)
            }
        default: break
        }
        // Nach jeder Aenderung Widget-Cache mit Debounce aktualisieren
        await WidgetDataWriter.shared.scheduleReload()
    }

    private func handleSummaryChange(_ change: AnyAction) async {
        // Weekly summary hat sich veraendert → Widget-Cache aktualisieren
        await WidgetDataWriter.shared.scheduleReload()
    }
}

// FamilyScore/Helpers/WidgetDataWriter.swift
// Target Membership: FamilyScore (App) ONLY — kein Widget-Extension-Import
// WidgetKit Import NUR fuer WidgetCenter.shared.reloadAllTimelines()

import Foundation
import WidgetKit
import FamilyScoreKit

@MainActor
final class WidgetDataWriter {
    static let shared = WidgetDataWriter()
    private init() {}

    // KRITISCH: UserDefaults(suiteName:) kann nil sein wenn App Group nicht konfiguriert
    // Pitfall SC-2 aus 05-RESEARCH.md: physisches Geraet in Wave 0 verifiziert
    private var defaults: UserDefaults? { UserDefaults(suiteName: appGroupIdentifier) }

    // Debounce-Task: verhindert zu viele reloadAllTimelines()-Aufrufe (iOS: 40-70/Tag Budget)
    private var debounceTask: Task<Void, Never>?

    /// Schreibt WidgetData sofort in App Group und reloadet Widget-Timeline.
    /// Fuer burst-artige Updates: scheduleReload() verwenden.
    func write(_ data: WidgetData) {
        guard let encoded = try? JSONEncoder().encode(data),
              let defaults = defaults else { return }
        defaults.set(encoded, forKey: "widgetData")
        // Kein synchronize() noetig — iOS schreibt automatisch
        WidgetCenter.shared.reloadAllTimelines()
    }

    /// Debounced Reload: wartet 5s nach letztem Aufruf, dann reloadAllTimelines().
    /// Verhindert iOS-Budget-Erschoepfung bei vielen Realtime-Events in kurzer Zeit.
    func scheduleReload() {
        debounceTask?.cancel()
        debounceTask = Task {
            try? await Task.sleep(nanoseconds: 5_000_000_000)  // 5 Sekunden
            if !Task.isCancelled {
                WidgetCenter.shared.reloadAllTimelines()
            }
        }
    }
}

// FamilyScoreWidgetExtension/Providers/FamilyScoreProvider.swift
// Target Membership: FamilyScoreWidgetExtension ONLY
// Kein Netzwerkaufruf — liest ausschliesslich aus App Group UserDefaults

import WidgetKit
import FamilyScoreKit

struct FamilyScoreProvider: TimelineProvider {

    func placeholder(in context: Context) -> FamilyScoreEntry {
        FamilyScoreEntry(date: Date(), data: .placeholder)
    }

    func getSnapshot(in context: Context,
                     completion: @escaping (FamilyScoreEntry) -> Void) {
        let data = WidgetData.readFromAppGroup() ?? .placeholder
        completion(FamilyScoreEntry(date: .now, data: data))
    }

    func getTimeline(in context: Context,
                     completion: @escaping (Timeline<FamilyScoreEntry>) -> Void) {
        let data = WidgetData.readFromAppGroup() ?? .placeholder
        let entry = FamilyScoreEntry(date: .now, data: data)
        // Stuendlicher Fallback-Refresh — normaler Refresh via WidgetCenter.shared.reloadAllTimelines()
        // nach jedem Realtime-Event (WidgetDataWriter.scheduleReload())
        let nextRefresh = Calendar.current.date(byAdding: .hour, value: 1, to: .now)!
        completion(Timeline(entries: [entry], policy: .after(nextRefresh)))
    }
}

        var int: UInt64 = 0
        Scanner(string: hex).scanHexInt64(&int)
        let r = Double((int & 0xFF0000) >> 16) / 255
        let g = Double((int & 0x00FF00) >> 8) / 255
